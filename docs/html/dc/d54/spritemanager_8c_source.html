26// Input: b=sprite height
27// a=x coordinate

36void ionPutSprite()__naked{
37 __asm
38
39 putSprite:
40 ld e,l
41 ld h,#00
42 ld d,h
43
44 // hl, de = 16 bit y. Must mult by X_LINE
45 Y_MULT ;Find the Y displacement offset
46
47
48 ld e,a
49 and #07 ;Find the bit number
50 ld c,a
51 srl e
52 srl e
53 srl e
54 add hl,de ;Find the X displacement offset
55 ld de,#plotSScreen
56 add hl,de
57 putSpriteLoop1:
58 sl1: ld d,(ix) ;loads image byte into D
59 ld e,#00
60 ld a,c
61 or a
62 jr z,putSpriteSkip1
63 putSpriteLoop2:
64 srl d ;rotate to give out smooth moving
65 rr e
66 dec a
67 jr nz,putSpriteLoop2
68 putSpriteSkip1:
69 ld a,(hl)
70 COPY_MODE d
71 ld (hl),a
72 inc hl
73 ld a,(hl)
74 COPY_MODE e
75 ld (hl),a ;copy to buffer using XOR logic
76 ld de,#X_LINE-1
77 add hl,de
78 inc ix ;Set for next byte of image
79 djnz putSpriteLoop1
80
81 ret
82
83 __endasm;
84}
